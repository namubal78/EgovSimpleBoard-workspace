DROP TABLE MEMBER CASCADE CONSTRAINTS;
DROP TABLE BOARD CASCADE CONSTRAINTS;
DROP TABLE REPLY CASCADE CONSTRAINTS;

DROP SEQUENCE SEQ_MEMBER_NO;
DROP SEQUENCE SEQ_BOARD_NO;
DROP SEQUENCE SEQ_REPLY_NO;

CREATE TABLE MEMBER (
    MEMBER_NO NUMBER PRIMARY KEY,
    MEMBER_ID VARCHAR2(20) NOT NULL,
    MEMBER_PWD VARCHAR2(30) NOT NULL,
    MEMBER_NAME VARCHAR2(10) NOT NULL,
    EMAIL VARCHAR2(40),
    PHONE VARCHAR2(13) NOT NULL,
    MEMBER_STATUS CHAR(1) DEFAULT 'Y' NOT NULL);
    
CREATE SEQUENCE SEQ_MEMBER_NO NOCACHE;

CREATE TABLE BOARD (
    BOARD_NO NUMBER PRIMARY KEY,
    BOARD_TITLE VARCHAR2(100) NOT NULL,
    BOARD_CONTENT VARCHAR2(3000),
    BOARD_WRITER VARCHAR2(30) NOT NULL,
    BOARD_DATE DATE DEFAULT SYSDATE,
    BOARD_COUNT NUMBER DEFAULT 1,
    ORIGIN_NAME VARCHAR2(255),
    CHANGE_NAME VARCHAR2(255),
    BOARD_STATUS CHAR(1) DEFAULT 'Y' NOT NULL,
    MEMBER_NO NUMBER NOT NULL);
    
CREATE SEQUENCE SEQ_BOARD_NO NOCACHE;

ALTER TABLE BOARD ADD FOREIGN KEY(MEMBER_NO) REFERENCES MEMBER(MEMBER_NO);

CREATE TABLE REPLY (
    REPLY_NO NUMBER PRIMARY KEY,
    REPLY_CONTENT VARCHAR2(3000) NOT NULL,
    REPLY_WRITER VARCHAR2(30) NOT NULL,
    REPLY_DATE DATE DEFAULT SYSDATE,
    REPLY_STATUS CHAR(1) DEFAULT 'Y' NOT NULL,
    BOARD_NO NUMBER NOT NULL);
    
CREATE SEQUENCE SEQ_REPLY_NO NOCACHE;

ALTER TABLE REPLY ADD FOREIGN KEY(BOARD_NO) REFERENCES BOARD(BOARD_NO);
